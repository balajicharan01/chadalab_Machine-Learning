3. Gradient Descent Implementation:

• Implement the gradient descent algorithm (without the Scikit-Learn library) to find the regression line. Initialize parameters randomly and update them iteratively to minimize the loss function. Record the loss value for each iteration:

In [4]:
import numpy as np
import matplotlib.pyplot as plt
from sklearn.metrics import mean_squared_error

# Initialize parameters
X_b = np.c_[np.ones((X_train.shape[0], 1)), X_train]
m = X_train.shape[0]  # number of data points
n = X_train.shape[1]  # number of features

n_iterations = 10000   # Number of iterations
W = np.random.randn(n + 1, 1)  # Weight matrix

# Gradient descent with constant learning rate
alpha = 0.01                  # Learning rate
loss_constant_lr = []         # Loss value for each iteration

for iteration in range(n_iterations):
    gradients = 2/m * X_b.T.dot(X_b.dot(W) - y_train.values.reshape(-1, 1))  # Convert y_train to 2D array
    W = W - alpha * gradients
    predictions = X_b.dot(W)
    loss_constant_lr.append(mean_squared_error(y_train, predictions))

# Gradient descent with decaying learning rate
alpha = 0.01                  # Initial learning rate
loss_decaying_lr = []         # Loss value for each iteration

for iteration in range(n_iterations):
    gradients = 2/m * X_b.T.dot(X_b.dot(W) - y_train.values.reshape(-1, 1))  # Convert y_train to 2D array
    alpha = alpha / (1 + iteration * 0.001)  # Exponential decay of learning rate
    W = W - alpha * gradients
    predictions = X_b.dot(W)
    loss_decaying_lr.append(mean_squared_error(y_train, predictions))

# Show the coefficients and intercept of the model
print("Coefficients:", W[1:])
print("Intercept:", W[0])

# Plot the loss values as a function of the number of iterations
plt.figure(figsize=(10, 6))
plt.plot(range(n_iterations), loss_constant_lr, label='Constant Learning Rate')
plt.plot(range(n_iterations), loss_decaying_lr, label='Decaying Learning Rate')
plt.title('Loss vs. Number of Iterations')
plt.xlabel('Number of Iterations')
plt.ylabel('Loss')
plt.legend()
plt.grid(True)
plt.show()

5. Learning Rate Analysis:

• Demonstrate the effect of varying the learning rate on the convergence of the gradient descent algorithm.

In [6]:
import numpy as np
import matplotlib.pyplot as plt
from sklearn.metrics import mean_squared_error

# Initialize parameters
X_b = np.c_[np.ones((X_train.shape[0], 1)), X_train]
m = X_train.shape[0]  # number of data points
n = X_train.shape[1]  # number of features

n_iterations = 1000   # Number of iterations
learning_rates = [0.001, 0.01, 0.1, 0.5]  # List of learning rates
losses = []

# Gradient descent with varying learning rates
for alpha in learning_rates:
    W = np.random.randn(n + 1, 1)  # Weight matrix
    loss_lr = []  # Loss value for each iteration
    
    for iteration in range(n_iterations):
        gradients = 2/m * X_b.T.dot(X_b.dot(W) - y_train.values.reshape(-1, 1))
        W = W - alpha * gradients
        predictions = X_b.dot(W)
        loss_lr.append(mean_squared_error(y_train, predictions))
    
    losses.append(loss_lr)

# Plot the loss values as a function of the number of iterations for each learning rate
plt.figure(figsize=(10, 6))
for i, alpha in enumerate(learning_rates):
    plt.plot(range(n_iterations), losses[i], label=f'Learning Rate: {alpha}')
plt.title('Loss vs. Number of Iterations for Different Learning Rates')
plt.xlabel('Number of Iterations')
plt.ylabel('Loss')
plt.legend()
plt.grid(True)
plt.show()

7. Normal Equation Implementation:

• Use the normal equation method to find the regression line directly.

• Compare the MAE and MSE with previous methods.

In [10]:
# Adding a bias term to the feature matrix (X_train)
X_train_bias = np.c_[np.ones((X_train.shape[0], 1)), X_train]

# Compute the normal equation to find regression coefficients
theta_normal_eq = np.linalg.inv(X_train_bias.T.dot(X_train_bias)).dot(X_train_bias.T).dot(y_train)

# Predict expenses for the testing dataset using the normal equation coefficients
X_test_bias = np.c_[np.ones((X_test.shape[0], 1)), X_test]
y_pred_normal_eq = X_test_bias.dot(theta_normal_eq)

# Compute Mean Absolute Error (MAE) and Mean Squared Error (MSE) for normal equation method
mae_normal_eq = mean_absolute_error(y_test, y_pred_normal_eq)
mse_normal_eq = mean_squared_error(y_test, y_pred_normal_eq)

print("Normal Equation Method:")
print("Mean Absolute Error (MAE):", mae_normal_eq)

# Compute error for Normal Equation Method
